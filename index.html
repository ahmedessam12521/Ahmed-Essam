VIEW Top Customers
Aggregates customer orders and payment amounts by month and year using multi-table joins across customers, orders, and payments.
Create View Top_Customers As
select c.CustomerNumber,
  customerName,
  count(o.orderNumber) As Orders_count,
  sum(p.amount) As Amounts,
  DATE_FORMAT(orderDate,'%Y') AS Year,
  DATE_FORMAT(orderDate,'%M') AS Month
From customers c
inner join orders o
  on c.customerNumber = o.customerNumber
inner join payments p
  on c.customerNumber = p.customerNumber
group by customerName, Year, Month,
  c.CustomerNumber
order by Year;
VIEW Product Analysis
Calculates total cost, revenue, and profit per product per month by joining products, order details, orders, customers, and payments.
Create View Products_Analysis As
select c.CustomerNumber,
  p.productName,
  sum(o.quantityOrdered) As Quantity_Ordered,
  sum(p.buyPrice*o.quantityOrdered)
    As Total_Cost,
  sum(o.priceEach*o.quantityOrdered)
    As Total_Revenue,
  sum(o.priceEach*o.quantityOrdered)
  - sum(p.buyPrice*o.quantityOrdered)
    As Total_Profit,
  DATE_FORMAT(orderDate,'%Y') AS Year,
  DATE_FORMAT(orderDate,'%M') AS Month
from products p
inner join orderdetails o
  on p.productCode = o.productCode
inner join orders os
  on o.orderNumber = os.orderNumber
inner join customers c
  on os.CustomerNumber = c.customerNumber
group by p.productName, Month, Year,
  c.CustomerNumber
order by Year;
VIEW Shipping Speed Classifier
Uses DATEDIFF and a CASE statement to classify each order as Express, Standard, or Delayed Delivery based on shipping duration.
Create View Orders_dates As
select customerNumber,
  coalesce(orderNumber) As Order_Number,
  orderDate,
  coalesce(shippedDate) As Shipping_Date,
  DATEDIFF(shippedDate,orderDate)
    As Duration,
  DATE_FORMAT(orderDate,'%Y') AS Year,
  DATE_FORMAT(orderDate,'%M') AS Month,
  Case
    When DATEDIFF(shippedDate,orderDate)
      <= 2 Then "Express Delivery"
    When DATEDIFF(shippedDate,orderDate)
      Between 3 and 5
      Then "Standard Delivery"
    Else "Delayed Delivery"
  END AS shipping_speed
from orders
where shippedDate IS NOT NULL
order by Year;
VIEW Churn Analysis
Classifies customers as Active, At Risk, or Churned using TIMESTAMPDIFF against a reference date, with HAVING to filter inactive accounts.
create view churned_customers As
Select c.customernumber,
  c.customerName,
  Max(o.orderDate) As last_order_date,
  timestampdiff(Month,
    least(Max(o.orderdate),'2005-04-22'),
    '2005-04-22')
    As months_since_last_order,
  CASE
    When timestampdiff(Month,
      least(Max(o.orderdate),'2005-04-22'),
      '2005-04-22') <= 3 then "Active"
    WHEN TIMESTAMPDIFF(Month,
      LEAST(MAX(o.orderDate),'2005-04-22'),
      '2005-04-22') BETWEEN 4 AND 6
      THEN 'At Risk'
    else "Churned"
  end As Churn_Status
from customers c
inner join orders o
  on c.customerNumber = o.customerNumber
group by c.customerNumber, c.customerName
HAVING MAX(o.orderDate) <
  DATE_SUB('2005-04-22', INTERVAL 6 MONTH)
order by months_since_last_order DESC;
CTE + WINDOW Top 3 Products Per Category
Uses two CTEs and DENSE_RANK() window function partitioned by category to find the top 3 best-selling products in each category.
WITH Product_Sales AS (
  SELECT p.ProductID, p.ProductName,
    p.CategoryID,
    SUM(od.Quantity * od.UnitPrice)
      AS TotalSales
  FROM products p
  INNER JOIN orderdetails od
    ON p.ProductID = od.ProductID
  GROUP BY p.ProductID, p.ProductName,
    p.CategoryID
),
RankedProducts AS (
  SELECT ps.ProductID, ps.ProductName,
    ps.CategoryID, ps.TotalSales,
    DENSE_RANK() OVER (
      PARTITION BY ps.CategoryID
      ORDER BY ps.TotalSales DESC
    ) AS rnk
  FROM Product_Sales ps
)
SELECT c.CategoryName, rp.ProductName,
  rp.TotalSales,
  rp.rnk AS RankInCategory
FROM RankedProducts rp
INNER JOIN categories c
  ON rp.CategoryID = c.CategoryID
WHERE rp.rnk <= 3
ORDER BY c.CategoryName, rp.rnk;
PROCEDURE Orders By Customer
A stored procedure that accepts a CustomerID parameter and returns all orders with amounts for that specific customer.
DELIMITER $$

CREATE PROCEDURE GetOrdersByCustomer(
  IN p_CustomerID INT
)
BEGIN
  SELECT
    CONCAT(c.FirstName,' ',c.LastName)
      AS FullName,
    o.OrderID,
    o.TotalAmount
  FROM orders o
  INNER JOIN customers c
    ON o.CustomerID = c.CustomerID
  WHERE c.CustomerID = p_CustomerID;
END$$

DELIMITER ;

call GetOrdersByCustomer(10);
QUERY Customer Ranking by Revenue
Ranks all customers by total order amount in 2024 using the RANK() window function, joined across orders, customers, and a calendar table.
select
  concat(c.FirstName," ",c.LastName)
    As FullName,
  sum(o.TotalAmount) As TotalAmount,
  cal.YEAR,
  Rank() over (
    Order by sum(o.TotalAmount) Desc
  ) As CustomerRank
from orders o
inner join customers c
  on o.CustomerID = c.CustomerID
inner join calendar cal
  on o.OrderDateID = cal.DateID
where cal.YEAR = 2024
group by
  concat(c.FirstName," ",c.LastName),
  cal.YEAR
order by TotalAmount DESC;
QUERY Above-Average Price Products
Uses a correlated subquery to find all products priced above the overall average, returning them with the average for comparison.
SELECT
  ProductName,
  Price,
  (SELECT AVG(Price) FROM products)
    AS AveragePrice
from products
WHERE Price > (
  SELECT AVG(Price) FROM products
)
order by Price Desc;
